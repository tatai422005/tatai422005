  <script>
    // === Game Config ===
    const CANVAS = document.getElementById('game');
    const CTX = CANVAS.getContext('2d');

    const GRID = 20; // 20x20 grid
    const CELL = CANVAS.width / GRID; // 30px per cell at 600px canvas

    const SPEEDS = [
      { label: 'Chill', stepMs: 160 },
      { label: 'Normal', stepMs: 110 },
      { label: 'Fast', stepMs: 80 },
      { label: 'Insane', stepMs: 55 }
    ];

    // === UI Elements ===
    const scoreEl = document.getElementById('score');
    const highEl = document.getElementById('high');
    const speedLabelEl = document.getElementById('speedLabel');
    const pauseBtn = document.getElementById('pauseBtn');
    const restartBtn = document.getElementById('restartBtn');
    const speedBtn = document.getElementById('speedBtn');

    // === Persistent High Score ===
    const HIGH_KEY = 'snake.highscore.v1';
    const getHigh = () => Number(localStorage.getItem(HIGH_KEY) || 0);
    const setHigh = v => localStorage.setItem(HIGH_KEY, String(v));

    // === Game State ===
    let snake, dir, nextDir, food, score, paused, over, lastTime, acc, speedIdx;

    function init() {
      snake = [ {x: 10, y: 10}, {x: 9, y: 10}, {x: 8, y: 10} ];
      dir = {x: 1, y: 0};
      nextDir = {x: 1, y: 0};
      food = spawnFood();
      score = 0; paused = false; over = false; lastTime = 0; acc = 0; speedIdx = 1; // Normal
      scoreEl.textContent = score;
      highEl.textContent = getHigh();
      speedLabelEl.textContent = SPEEDS[speedIdx].label;
      requestAnimationFrame(loop);
    }

    function spawnFood() {
      let pos;
      do {
        pos = { x: Math.floor(Math.random() * GRID), y: Math.floor(Math.random() * GRID) };
      } while (snake.some(s => s.x === pos.x && s.y === pos.y));
      return pos;
    }

    function loop(ts) {
      if (over) return draw();
      const dt = ts - lastTime; lastTime = ts; acc += dt;
      const stepMs = SPEEDS[speedIdx].stepMs;
      if (!paused) {
        while (acc >= stepMs) { step(); acc -= stepMs; }
      }
      draw();
      requestAnimationFrame(loop);
    }

    function step() {
      // apply queued direction
      dir = nextDir;
      const head = { x: snake[0].x + dir.x, y: snake[0].y + dir.y };

      // Wall collision => game over
      if (head.x < 0 || head.y < 0 || head.x >= GRID || head.y >= GRID) {
        over = true; return;
      }

      // Self collision
      if (snake.some(s => s.x === head.x && s.y === head.y)) { over = true; return; }

      snake.unshift(head);

      // Food
      if (head.x === food.x && head.y === food.y) {
        score += 1; scoreEl.textContent = score;
        if (score > getHigh()) { setHigh(score); highEl.textContent = score; }
        food = spawnFood();
      } else {
        snake.pop(); // move
      }
    }

    function drawCell(x, y, fill, stroke) {
      const px = x * CELL, py = y * CELL;
      CTX.fillStyle = fill; CTX.fillRect(px, py, CELL, CELL);
      if (stroke) { CTX.strokeStyle = stroke; CTX.strokeRect(px + 0.5, py + 0.5, CELL - 1, CELL - 1); }
    }

    function draw() {
      // background grid
      CTX.clearRect(0, 0, CANVAS.width, CANVAS.height);

      // food glow
      drawCell(food.x, food.y, '#2ee6a7', '#1ab383');

      // snake
      for (let i = snake.length - 1; i >= 0; i--) {
        const s = snake[i];
        const isHead = i === 0;
        drawCell(s.x, s.y, isHead ? '#6ee7ff' : '#59b7ff', 'rgba(0,0,0,0.25)');
        if (isHead) {
          // eyes
          const cx = s.x * CELL + CELL / 2, cy = s.y * CELL + CELL / 2;
          CTX.fillStyle = '#0b0e1a';
          CTX.beginPath(); CTX.arc(cx - 5, cy - 5, 2.2, 0, Math.PI * 2); CTX.fill();
          CTX.beginPath(); CTX.arc(cx + 5, cy - 5, 2.2, 0, Math.PI * 2); CTX.fill();
        }
      }

      // overlays
      if (paused) overlayText('Paused ‚Äî Press Space');
      if (over) overlayText('Game Over ‚Äî Press R');
    }

    function overlayText(text) {
      CTX.save();
      CTX.fillStyle = 'rgba(0,0,0,0.45)';
      CTX.fillRect(0, 0, CANVAS.width, CANVAS.height);
      CTX.fillStyle = '#fff';
      CTX.font = '700 28px system-ui, -apple-system, Segoe UI, Roboto';
      CTX.textAlign = 'center';
      CTX.fillText(text, CANVAS.width / 2, CANVAS.height / 2);
      CTX.restore();
    }

    // === Input ===
    const queueDir = (nx, ny) => {
      // disallow reversing directly
      if (nx === -dir.x && ny === -dir.y) return;
      nextDir = { x: nx, y: ny };
    };

    window.addEventListener('keydown', (e) => {
      const k = e.key.toLowerCase();
      if (k === 'arrowleft' || k === 'a') queueDir(-1, 0);
      else if (k === 'arrowright' || k === 'd') queueDir(1, 0);
      else if (k === 'arrowup' || k === 'w') queueDir(0, -1);
      else if (k === 'arrowdown' || k === 's') queueDir(0, 1);
      else if (k === ' ') { paused = !paused; }
      else if (k === 'r') restart();
    });

    pauseBtn.addEventListener('click', () => paused = !paused);
    restartBtn.addEventListener('click', restart);
    speedBtn.addEventListener('click', () => {
      speedIdx = (speedIdx + 1) % SPEEDS.length;
      speedLabelEl.textContent = SPEEDS[speedIdx].label;
    });

    // D-pad
    document.querySelectorAll('.dkey').forEach(btn => {
      btn.addEventListener('click', () => {
        const d = btn.getAttribute('data-dir');
        if (d === 'left') queueDir(-1, 0);
        if (d === 'right') queueDir(1, 0);
        if (d === 'up') queueDir(0, -1);
        if (d === 'down') queueDir(0, 1);
      });
    });

    function restart() {
      const keepSpeed = speedIdx; // preserve chosen speed
      init();
      speedIdx = keepSpeed;
      speedLabelEl.textContent = SPEEDS[speedIdx].label;
    }

    // Start game
    init();
  </script>
